def create_walk_forward_splits(data: pd.DataFrame,

    """

    """–ê–±—Å—Ç—Ä–∞–∫—Ç–Ω—ã–π –±–∞–∑–æ–≤—ã–π –∫–ª–∞—Å—Å –¥–ª—è —Ç–æ—Ä–≥–æ–≤—ã—Ö —Å—Ç—Ä–∞—Ç–µ–≥–∏–π"""

class SimpleMovingAverageStrategy(TradingStrategy):
    """–ü—Ä–æ—Å—Ç–∞—è —Å—Ç—Ä–∞—Ç–µ–≥–∏—è –Ω–∞ –æ—Å–Ω–æ–≤–µ —Å–∫–æ–ª—å–∑—è—â–∏—Ö —Å—Ä–µ–¥–Ω–∏—Ö"""
    
    def __init__(self, short_window: int = 20, long_window: int = 50):
        self.short_window = short_window
        self.long_window = long_window
        self.short_ma = None
        self.long_ma = None
        self.is_trained = False
    
    def train(self, data: pd.DataFrame) -> None:
        """–û–±—É—á–µ–Ω–∏–µ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏ (–≤ –¥–∞–Ω–Ω–æ–º —Å–ª—É—á–∞–µ –ø—Ä–æ—Å—Ç–æ —Ä–∞—Å—á–µ—Ç –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤)"""
        if 'close' not in data.columns:
            raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")
        
        self.short_ma = data['close'].rolling(window=self.short_window).mean()
        self.long_ma = data['close'].rolling(window=self.long_window).mean()
        self.is_trained = True
    
    def predict(self, data: pd.DataFrame) -> pd.Series:
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Ç–æ—Ä–≥–æ–≤—ã—Ö —Å–∏–≥–Ω–∞–ª–æ–≤"""
        if not self.is_trained:
            raise ValueError("–°—Ç—Ä–∞—Ç–µ–≥–∏—è –Ω–µ –æ–±—É—á–µ–Ω–∞. –í—ã–∑–æ–≤–∏—Ç–µ train() —Å–Ω–∞—á–∞–ª–∞.")
        
        if 'close' not in data.columns:
            raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")
        
        # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º —Å–∫–æ–ª—å–∑—è—â–∏–µ —Å—Ä–µ–¥–Ω–∏–µ –¥–ª—è –Ω–æ–≤—ã—Ö –¥–∞–Ω–Ω—ã—Ö
        short_ma = data['close'].rolling(window=self.short_window).mean()
        long_ma = data['close'].rolling(window=self.long_window).mean()
        
        # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Å–∏–≥–Ω–∞–ª—ã: 1 = –ø–æ–∫—É–ø–∫–∞, -1 = –ø—Ä–æ–¥–∞–∂–∞, 0 = —É–¥–µ—Ä–∂–∞–Ω–∏–µ
        signals = pd.Series(0, index=data.index)
        
        # –°–∏–≥–Ω–∞–ª –ø–æ–∫—É–ø–∫–∏: –∫–æ—Ä–æ—Ç–∫–∞—è MA –ø–µ—Ä–µ—Å–µ–∫–∞–µ—Ç –¥–ª–∏–Ω–Ω—É—é MA —Å–Ω–∏–∑—É –≤–≤–µ—Ä—Ö
        buy_signal = (short_ma > long_ma) & (short_ma.shift(1) <= long_ma.shift(1))
        
        # –°–∏–≥–Ω–∞–ª –ø—Ä–æ–¥–∞–∂–∏: –∫–æ—Ä–æ—Ç–∫–∞—è MA –ø–µ—Ä–µ—Å–µ–∫–∞–µ—Ç –¥–ª–∏–Ω–Ω—É—é MA —Å–≤–µ—Ä—Ö—É –≤–Ω–∏–∑
        sell_signal = (short_ma < long_ma) & (short_ma.shift(1) >= long_ma.shift(1))
        
        signals[buy_signal] = 1
        signals[sell_signal] = -1
        
        return signals
    
    def get_name(self) -> str:
        return f"SMA_{self.short_window}_{self.long_window}"

class Backtester:
    """–ö–ª–∞—Å—Å –¥–ª—è –ø—Ä–æ–≤–µ–¥–µ–Ω–∏—è –±—ç–∫—Ç–µ—Å—Ç–∏–Ω–≥–∞ —Ç–æ—Ä–≥–æ–≤—ã—Ö —Å—Ç—Ä–∞—Ç–µ–≥–∏–π"""
    
    def __init__(self, initial_capital: float = 100000.0, commission: float = 0.001):
        self.initial_capital = initial_capital
        self.commission = commission
    
    def run_backtest(self, data: pd.DataFrame, strategy: TradingStrategy) -> Dict[str, float]:
        """
        –ó–∞–ø—É—Å–∫ –±—ç–∫—Ç–µ—Å—Ç–∏–Ω–≥–∞ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏
        
        Args:
            data: –î–∞–Ω–Ω—ã–µ –¥–ª—è —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è
            strategy: –û–±—É—á–µ–Ω–Ω–∞—è —Ç–æ—Ä–≥–æ–≤–∞—è —Å—Ç—Ä–∞—Ç–µ–≥–∏—è
            
        Returns:
            –°–ª–æ–≤–∞—Ä—å —Å –º–µ—Ç—Ä–∏–∫–∞–º–∏ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
        """
        if 'close' not in data.columns:
            raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")
        
        # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Ç–æ—Ä–≥–æ–≤—ã–µ —Å–∏–≥–Ω–∞–ª—ã
        signals = strategy.predict(data)
        
        # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç—å
        returns = data['close'].pct_change()
        
        # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º —Å—Ç—Ä–∞—Ç–µ–≥–∏—á–µ—Å–∫—É—é –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç—å
        strategy_returns = signals.shift(1) * returns  # –°–¥–≤–∏–≥–∞–µ–º —Å–∏–≥–Ω–∞–ª—ã –Ω–∞ 1 –ø–µ—Ä–∏–æ–¥
        
        # –£—á–∏—Ç—ã–≤–∞–µ–º –∫–æ–º–∏—Å—Å–∏—é
        position_changes = signals.diff().abs()
        strategy_returns -= position_changes * self.commission
        
        # –£–¥–∞–ª—è–µ–º NaN –∑–Ω–∞—á–µ–Ω–∏—è
        strategy_returns = strategy_returns.dropna()
        
        if len(strategy_returns) == 0:
            return {
                'total_return': 0.0,
                'sharpe_ratio': 0.0,
                'max_drawdown': 0.0,
                'win_rate': 0.0,
                'total_trades': 0
            }
        
        # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º –∫—É–º—É–ª—è—Ç–∏–≤–Ω—É—é –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç—å
        cumulative_returns = (1 + strategy_returns).cumprod()
        
        # –û—Å–Ω–æ–≤–Ω—ã–µ –º–µ—Ç—Ä–∏–∫–∏
        total_return = cumulative_returns.iloc[-1] - 1
        
        # Sharpe Ratio (–ø—Ä–µ–¥–ø–æ–ª–∞–≥–∞–µ–º –±–µ–∑—Ä–∏—Å–∫–æ–≤—É—é —Å—Ç–∞–≤–∫—É = 0)
        sharpe_ratio = strategy_returns.mean() / strategy_returns.std() * np.sqrt(252) if strategy_returns.std() > 0 else 0
        
        # –ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –ø—Ä–æ—Å–∞–¥–∫–∞
        rolling_max = cumulative_returns.expanding().max()
        drawdowns = (cumulative_returns - rolling_max) / rolling_max
        max_drawdown = drawdowns.min()
        
        # –ü—Ä–æ—Ü–µ–Ω—Ç –≤—ã–∏–≥—Ä—ã—à–Ω—ã—Ö —Å–¥–µ–ª–æ–∫
        winning_trades = strategy_returns[strategy_returns > 0]
        total_trades = len(strategy_returns[strategy_returns != 0])
        win_rate = len(winning_trades) / total_trades if total_trades > 0 else 0
        
        return {
            'total_return': total_return,
            'sharpe_ratio': sharpe_ratio,
            'max_drawdown': max_drawdown,
            'win_rate': win_rate,
            'total_trades': total_trades,
            'strategy_returns': strategy_returns,
            'cumulative_returns': cumulative_returns
        }

class WalkForwardAnalyzer:

        """
        –ê–Ω–∞–ª–∏–∑ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ Walk-Forward —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è.
        
        –≠—Ç–æ—Ç –º–µ—Ç–æ–¥ –ø—Ä–æ–≤–æ–¥–∏—Ç —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏–π –∞–Ω–∞–ª–∏–∑ –≤—Å–µ—Ö —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
        —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è, –≤—ã—á–∏—Å–ª—è—è –∫–ª—é—á–µ–≤—ã–µ –º–µ—Ç—Ä–∏–∫–∏ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
        –∏ —Å—Ç–∞–±–∏–ª—å–Ω–æ—Å—Ç–∏ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏.
        
        Returns:
            –°–ª–æ–≤–∞—Ä—å —Å —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞–º–∏ –∞–Ω–∞–ª–∏–∑–∞
        """
        if not self.results:
            raise ValueError("–ù–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ –¥–ª—è –∞–Ω–∞–ª–∏–∑–∞. –°–Ω–∞—á–∞–ª–∞ –∑–∞–ø—É—Å—Ç–∏—Ç–µ run_analysis()")
        
        # –ò–∑–≤–ª–µ—á–µ–Ω–∏–µ –º–µ—Ç—Ä–∏–∫ –∏–∑ –≤—Å–µ—Ö –ø–µ—Ä–∏–æ–¥–æ–≤
        returns = [r['metrics']['total_return'] for r in self.results]
        sharpe_ratios = [r['metrics']['sharpe_ratio'] for r in self.results]
        max_drawdowns = [r['metrics']['max_drawdown'] for r in self.results]
        win_rates = [r['metrics']['win_rate'] for r in self.results]
        total_trades = [r['metrics']['total_trades'] for r in self.results]
        
        # –ö–æ–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ–º –≤ numpy –º–∞—Å—Å–∏–≤—ã –¥–ª—è —É–¥–æ–±—Å—Ç–≤–∞ –≤—ã—á–∏—Å–ª–µ–Ω–∏–π
        returns = np.array(returns)
        sharpe_ratios = np.array(sharpe_ratios)
        max_drawdowns = np.array(max_drawdowns)
        win_rates = np.array(win_rates)
        total_trades = np.array(total_trades)
        
        # –û—Å–Ω–æ–≤–Ω–∞—è —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
        analysis = {
            # –û–±—â–∞—è –∏–Ω—Ñ–æ—Ä–º–∞—Ü–∏—è
            'total_periods': len(self.results),
            'successful_periods': len([r for r in returns if not np.isnan(r)]),
            
            # –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç–∏
            'mean_return': np.nanmean(returns),
            'std_return': np.nanstd(returns),
            'min_return': np.nanmin(returns),
            'max_return': np.nanmax(returns),
            'median_return': np.nanmedian(returns),
            
            # –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ Sharpe Ratio
            'mean_sharpe': np.nanmean(sharpe_ratios),
            'std_sharpe': np.nanstd(sharpe_ratios),
            'min_sharpe': np.nanmin(sharpe_ratios),
            'max_sharpe': np.nanmax(sharpe_ratios),
            
            # –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ –ø—Ä–æ—Å–∞–¥–æ–∫
            'mean_drawdown': np.nanmean(max_drawdowns),
            'worst_drawdown': np.nanmin(max_drawdowns),
            'std_drawdown': np.nanstd(max_drawdowns),
            
            # –°—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ —Ç–æ—Ä–≥–æ–≤–ª–∏
            'mean_win_rate': np.nanmean(win_rates),
            'mean_trades_per_period': np.nanmean(total_trades),
            'total_trades': np.nansum(total_trades),
            
            # –ö–æ–Ω—Å–∏—Å—Ç–µ–Ω—Ç–Ω–æ—Å—Ç—å
            'positive_periods': np.sum(returns > 0),
            'negative_periods': np.sum(returns < 0),
            'consistency': np.sum(returns > 0) / len(returns) if len(returns) > 0 else 0,
            
            # –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ –º–µ—Ç—Ä–∏–∫–∏
            'coefficient_of_variation': np.nanstd(returns) / np.abs(np.nanmean(returns)) if np.nanmean(returns) != 0 else np.inf,
            'skewness': self._calculate_skewness(returns),
            'kurtosis': self._calculate_kurtosis(returns),
            
            # –î–µ—Ç–∞–ª—å–Ω—ã–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã
            'period_returns': returns.tolist(),
            'period_sharpe_ratios': sharpe_ratios.tolist(),
            'period_drawdowns': max_drawdowns.tolist()
        }
        
        return analysis
    
    def _calculate_skewness(self, data: np.ndarray) -> float:
        """–†–∞—Å—á–µ—Ç –∞—Å–∏–º–º–µ—Ç—Ä–∏–∏ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è"""
        data_clean = data[~np.isnan(data)]
        if len(data_clean) < 3:
            return 0.0
        
        mean = np.mean(data_clean)
        std = np.std(data_clean)
        if std == 0:
            return 0.0
        
        return np.mean(((data_clean - mean) / std) ** 3)
    
    def _calculate_kurtosis(self, data: np.ndarray) -> float:
        """–†–∞—Å—á–µ—Ç —ç–∫—Å—Ü–µ—Å—Å–∞ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è"""
        data_clean = data[~np.isnan(data)]
        if len(data_clean) < 4:
            return 0.0
        
        mean = np.mean(data_clean)
        std = np.std(data_clean)
        if std == 0:
            return 0.0
        
        return np.mean(((data_clean - mean) / std) ** 4) - 3
```

def adaptive_walk_forward(data: pd.DataFrame, 
                         strategy: TradingStrategy, 
                         min_train: int = 126,

    —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏. –≠—Ç–æ –ø–æ–∑–≤–æ–ª—è–µ—Ç —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏ –ª—É—á—à–µ –∞–¥–∞–ø—Ç–∏—Ä–æ–≤–∞—Ç—å—Å—è –∫ –∏–∑–º–µ–Ω–µ–Ω–∏—è–º
    —Ä—ã–Ω–æ—á–Ω—ã—Ö —É—Å–ª–æ–≤–∏–π.
    
    Args:

        strategy: –¢–æ—Ä–≥–æ–≤–∞—è —Å—Ç—Ä–∞—Ç–µ–≥–∏—è –¥–ª—è —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è
        min_train: –ú–∏–Ω–∏–º–∞–ª—å–Ω—ã–π —Ä–∞–∑–º–µ—Ä –æ–±—É—á–∞—é—â–µ–≥–æ –æ–∫–Ω–∞
        max_train: –ú–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —Ä–∞–∑–º–µ—Ä –æ–±—É—á–∞—é—â–µ–≥–æ –æ–∫–Ω–∞

        performance_threshold_high: –ü–æ—Ä–æ–≥ –¥–ª—è —É–≤–µ–ª–∏—á–µ–Ω–∏—è –æ–∫–Ω–∞
        performance_threshold_low: –ü–æ—Ä–æ–≥ –¥–ª—è —É–º–µ–Ω—å—à–µ–Ω–∏—è –æ–∫–Ω–∞

        test_data = data.iloc[start_idx + current_train_size:start_idx + current_train_size + test_size]

              f"{test_data.index[-1].strftime('%Y-%m-%d')}")
        
        try:
            # –û–±—É—á–µ–Ω–∏–µ –∏ —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ
            strategy.train(train_data)

            total_return = metrics['total_return']

                current_train_size = min(current_train_size + window_adjustment, max_train)
                adjustment_reason = "—É–≤–µ–ª–∏—á–µ–Ω–∏–µ (—Ö–æ—Ä–æ—à–∞—è –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å)"
            elif total_return < performance_threshold_low:
                # –ü–ª–æ—Ö–∞—è –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å - —É–º–µ–Ω—å—à–∞–µ–º –æ–∫–Ω–æ
                current_train_size = max(current_train_size - window_adjustment, min_train)
                adjustment_reason = "—É–º–µ–Ω—å—à–µ–Ω–∏–µ (–ø–ª–æ—Ö–∞—è –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç—å)"
            else:

            # –°–æ—Ö—Ä–∞–Ω—è–µ–º —Ä–µ–∑—É–ª—å—Ç–∞—Ç
            result = {

                'train_size': old_train_size,
                'new_train_size': current_train_size,
                'adjustment_reason': adjustment_reason,
                'test_start': test_data.index[0],
                'test_end': test_data.index[-1],
                'metrics': metrics
            }
            
            results.append(result)
            
            print(f"  –†–µ–∑—É–ª—å—Ç–∞—Ç: –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç—å={total_return:.2%}, "
                  f"Sharpe={metrics['sharpe_ratio']:.2f}, "
                  f"–æ–∫–Ω–æ: {old_train_size} ‚Üí {current_train_size} ({adjustment_reason})")
            
        except Exception as e:

            # –ü—Ä–æ–¥–æ–ª–∂–∞–µ–º —Å —Ç–µ–∫—É—â–∏–º —Ä–∞–∑–º–µ—Ä–æ–º –æ–∫–Ω–∞

        period += 1
    
    print(f"–ê–¥–∞–ø—Ç–∏–≤–Ω—ã–π –∞–Ω–∞–ª–∏–∑ –∑–∞–≤–µ—Ä—à–µ–Ω. –û–±—Ä–∞–±–æ—Ç–∞–Ω–æ {len(results)} –ø–µ—Ä–∏–æ–¥–æ–≤")
    print(f"–§–∏–Ω–∞–ª—å–Ω—ã–π —Ä–∞–∑–º–µ—Ä –æ–∫–Ω–∞: {current_train_size} –¥–Ω–µ–π")
    
    return results

def compare_strategies(results: Dict[str, Dict[str, Any]]) -> List[Tuple[str, Dict[str, float]]]:
    """
    –°—Ä–∞–≤–Ω–µ–Ω–∏–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã—Ö —Å—Ç—Ä–∞—Ç–µ–≥–∏–π.
    
    –≠—Ç–∞ —Ñ—É–Ω–∫—Ü–∏—è —Å—Ä–∞–≤–Ω–∏–≤–∞–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã Walk-Forward –∞–Ω–∞–ª–∏–∑–∞
    –¥–ª—è —Ä–∞–∑–Ω—ã—Ö —Å—Ç—Ä–∞—Ç–µ–≥–∏–π –∏ —Ä–∞–Ω–∂–∏—Ä—É–µ—Ç –∏—Ö –ø–æ –∫–ª—é—á–µ–≤—ã–º –º–µ—Ç—Ä–∏–∫–∞–º.
    
    Args:
        results: –°–ª–æ–≤–∞—Ä—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ –∞–Ω–∞–ª–∏–∑–∞ —Å—Ç—Ä–∞—Ç–µ–≥–∏–π
        
    Returns:
        –û—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫ —Å—Ç—Ä–∞—Ç–µ–≥–∏–π —Å –º–µ—Ç—Ä–∏–∫–∞–º–∏
    """
    
    comparison = {}
    
    for strategy_name, analysis in results.items():
        if analysis is None:
            continue
            
        comparison[strategy_name] = {
            'mean_return': analysis['mean_return'],
            'consistency': analysis['consistency'],
            'mean_sharpe': analysis['mean_sharpe'],
            'worst_drawdown': analysis['worst_drawdown'],
            'coefficient_of_variation': analysis['coefficient_of_variation'],
            'total_periods': analysis['total_periods']
        }
    
    # –°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ –ø–æ —Å—Ä–µ–¥–Ω–µ–π –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç–∏ (–ø–æ —É–±—ã–≤–∞–Ω–∏—é)
    sorted_strategies = sorted(
        comparison.items(), 
        key=lambda x: x[1]['mean_return'], 
        reverse=True
    )
    
    return sorted_strategies

class RSIStrategy(TradingStrategy):
    """–°—Ç—Ä–∞—Ç–µ–≥–∏—è –Ω–∞ –æ—Å–Ω–æ–≤–µ RSI –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–∞"""
    
    def __init__(self, rsi_period: int = 14, oversold: float = 30, overbought: float = 70):
        self.rsi_period = rsi_period
        self.oversold = oversold
        self.overbought = overbought
        self.is_trained = False
    
    def train(self, data: pd.DataFrame) -> None:
        """–û–±—É—á–µ–Ω–∏–µ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏"""
        if 'close' not in data.columns:
            raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")
        self.is_trained = True
    
    def predict(self, data: pd.DataFrame) -> pd.Series:
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Å–∏–≥–Ω–∞–ª–æ–≤ –Ω–∞ –æ—Å–Ω–æ–≤–µ RSI"""
        if not self.is_trained:
            raise ValueError("–°—Ç—Ä–∞—Ç–µ–≥–∏—è –Ω–µ –æ–±—É—á–µ–Ω–∞")
        
        if 'close' not in data.columns:
            raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")
        
        # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º RSI
        delta = data['close'].diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=self.rsi_period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=self.rsi_period).mean()
        rs = gain / loss
        rsi = 100 - (100 / (1 + rs))
        
        # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Å–∏–≥–Ω–∞–ª—ã
        signals = pd.Series(0, index=data.index)
        
        # –°–∏–≥–Ω–∞–ª –ø–æ–∫—É–ø–∫–∏: RSI –≤—ã—Ö–æ–¥–∏—Ç –∏–∑ –∑–æ–Ω—ã –ø–µ—Ä–µ–ø—Ä–æ–¥–∞–Ω–Ω–æ—Å—Ç–∏
        buy_signal = (rsi > self.oversold) & (rsi.shift(1) <= self.oversold)
        
        # –°–∏–≥–Ω–∞–ª –ø—Ä–æ–¥–∞–∂–∏: RSI –≤—ã—Ö–æ–¥–∏—Ç –∏–∑ –∑–æ–Ω—ã –ø–µ—Ä–µ–∫—É–ø–ª–µ–Ω–Ω–æ—Å—Ç–∏
        sell_signal = (rsi < self.overbought) & (rsi.shift(1) >= self.overbought)
        
        signals[buy_signal] = 1
        signals[sell_signal] = -1
        
        return signals
    
    def get_name(self) -> str:
        return f"RSI_{self.rsi_period}_{self.oversold}_{self.overbought}"

class BollingerBandsStrategy(TradingStrategy):
    """–°—Ç—Ä–∞—Ç–µ–≥–∏—è –Ω–∞ –æ—Å–Ω–æ–≤–µ –ø–æ–ª–æ—Å –ë–æ–ª–ª–∏–Ω–¥–∂–µ—Ä–∞"""
    
    def __init__(self, period: int = 20, std_dev: float = 2.0):
        self.period = period
        self.std_dev = std_dev
        self.is_trained = False
    
    def train(self, data: pd.DataFrame) -> None:
        """–û–±—É—á–µ–Ω–∏–µ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏"""
        if 'close' not in data.columns:
            raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")
        self.is_trained = True
    
    def predict(self, data: pd.DataFrame) -> pd.Series:
        """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Å–∏–≥–Ω–∞–ª–æ–≤ –Ω–∞ –æ—Å–Ω–æ–≤–µ –ø–æ–ª–æ—Å –ë–æ–ª–ª–∏–Ω–¥–∂–µ—Ä–∞"""
        if not self.is_trained:
            raise ValueError("–°—Ç—Ä–∞—Ç–µ–≥–∏—è –Ω–µ –æ–±—É—á–µ–Ω–∞")
        
        if 'close' not in data.columns:
            raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")
        
        # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º –ø–æ–ª–æ—Å—ã –ë–æ–ª–ª–∏–Ω–¥–∂–µ—Ä–∞
        sma = data['close'].rolling(window=self.period).mean()
        std = data['close'].rolling(window=self.period).std()
        upper_band = sma + (std * self.std_dev)
        lower_band = sma - (std * self.std_dev)
        
        # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Å–∏–≥–Ω–∞–ª—ã
        signals = pd.Series(0, index=data.index)
        
        # –°–∏–≥–Ω–∞–ª –ø–æ–∫—É–ø–∫–∏: —Ü–µ–Ω–∞ –∫–∞—Å–∞–µ—Ç—Å—è –Ω–∏–∂–Ω–µ–π –ø–æ–ª–æ—Å—ã
        buy_signal = (data['close'] <= lower_band) & (data['close'].shift(1) > lower_band.shift(1))
        
        # –°–∏–≥–Ω–∞–ª –ø—Ä–æ–¥–∞–∂–∏: —Ü–µ–Ω–∞ –∫–∞—Å–∞–µ—Ç—Å—è –≤–µ—Ä—Ö–Ω–µ–π –ø–æ–ª–æ—Å—ã
        sell_signal = (data['close'] >= upper_band) & (data['close'].shift(1) < upper_band.shift(1))
        
        signals[buy_signal] = 1
        signals[sell_signal] = -1
        
        return signals
    
    def get_name(self) -> str:
        return f"BB_{self.period}_{self.std_dev}"
```

def rolling_walk_forward(data, strategy, window_size=252, test_size=63):
    """Rolling Window Walk-Forward"""
    
    results = []
    start_idx = 0
    
    while start_idx + window_size + test_size <= len(data):
        # –û–±—É—á–∞—é—â–∏–π –ø–µ—Ä–∏–æ–¥ (—Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –æ–∫–Ω–æ)
        train_data = data.iloc[start_idx:start_idx + window_size]
        
        # –¢–µ—Å—Ç–æ–≤—ã–π –ø–µ—Ä–∏–æ–¥
        test_data = data.iloc[start_idx + window_size:start_idx + window_size + test_size]
        
        # –û–±—É—á–µ–Ω–∏–µ –∏ —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ
        strategy.train(train_data)
        backtester = Backtester()
        metrics = backtester.run_backtest(test_data, strategy)
        
        results.append(metrics)
        start_idx += test_size
    
    return results

def expanding_walk_forward(data, strategy, min_train=126, test_size=63):
    """Expanding Window Walk-Forward"""
    
    results = []
    start_idx = 0
    train_size = min_train
    
    while start_idx + train_size + test_size <= len(data):
        # –û–±—É—á–∞—é—â–∏–π –ø–µ—Ä–∏–æ–¥ (—Ä–∞—Å—à–∏—Ä—è—é—â–µ–µ—Å—è –æ–∫–Ω–æ)
        train_data = data.iloc[:start_idx + train_size]
        
        # –¢–µ—Å—Ç–æ–≤—ã–π –ø–µ—Ä–∏–æ–¥
        test_data = data.iloc[start_idx + train_size:start_idx + train_size + test_size]
        
        # –û–±—É—á–µ–Ω–∏–µ –∏ —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ
        strategy.train(train_data)
        backtester = Backtester()
        metrics = backtester.run_backtest(test_data, strategy)
        
        results.append(metrics)
        start_idx += test_size
        train_size += test_size  # –†–∞—Å—à–∏—Ä—è–µ–º –æ–∫–Ω–æ
    
    return results

```python
def analyze_stability(results):
    """–ê–Ω–∞–ª–∏–∑ —Å—Ç–∞–±–∏–ª—å–Ω–æ—Å—Ç–∏ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤"""
    
    returns = [r['metrics']['total_return'] for r in results]
    
    # –ö–æ—ç—Ñ—Ñ–∏—Ü–∏–µ–Ω—Ç –≤–∞—Ä–∏–∞—Ü–∏–∏
    cv = np.std(returns) / np.abs(np.mean(returns))
    
    # –¢—Ä–µ–Ω–¥ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
    x = np.arange(len(returns))
    slope, intercept, r_value, p_value, std_err = stats.linregress(x, returns)
    
    # –°—Ç–∞–±–∏–ª—å–Ω–æ—Å—Ç—å Sharpe Ratio
    sharpe_ratios = [r['metrics']['sharpe_ratio'] for r in results]
    sharpe_stability = 1 - np.std(sharpe_ratios) / np.abs(np.mean(sharpe_ratios))
    
    return {
        'coefficient_of_variation': cv,
        'performance_trend': slope,
        'trend_significance': p_value,
        'sharpe_stability': sharpe_stability,
        'return_consistency': 1 - cv
    }
```

```python
def analyze_degradation(results):
    """–ê–Ω–∞–ª–∏–∑ –¥–µ–≥—Ä–∞–¥–∞—Ü–∏–∏ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏"""
    
    returns = [r['metrics']['total_return'] for r in results]
    
    # –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ –Ω–∞ –ø–µ—Ä–∏–æ–¥—ã
    n_periods = len(returns)
    first_half = returns[:n_periods//2]
    second_half = returns[n_periods//2:]
    
    # –°—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
    first_half_mean = np.mean(first_half)
    second_half_mean = np.mean(second_half)
    
    degradation = (second_half_mean - first_half_mean) / abs(first_half_mean)
    
    # –°—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∏–π —Ç–µ—Å—Ç
    t_stat, p_value = stats.ttest_ind(first_half, second_half)
    
    return {
        'first_half_mean': first_half_mean,
        'second_half_mean': second_half_mean,
        'degradation': degradation,
        't_statistic': t_stat,
        'p_value': p_value,
        'significant_degradation': p_value < 0.05 and degradation < -0.1
    }
```
