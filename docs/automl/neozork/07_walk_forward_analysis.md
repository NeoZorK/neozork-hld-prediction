class SimpleMovingAverageStrategy(TradingStrategy):
 """–ü—Ä–æ—Å—Ç–∞—è —Å—Ç—Ä–∞—Ç–µ–≥–∏—è on basis —Å–∫–æ–ª—å–∑—è—â–∏—Ö —Å—Ä–µ–¥–Ω–∏—Ö"""

 def __init__(self, short_window: int = 20, long_window: int = 50):
 self.short_window = short_window
 self.long_window = long_window
 self.short_ma = None
 self.long_ma = None
 self.is_trained = False

 def train(self, data: pd.DataFrame) -> None:
 """–û–±—É—á–µ–Ω–∏–µ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏ (in –¥–∞–Ω–Ω–æ–º —Å–ª—É—á–∞–µ –ø—Ä–æ—Å—Ç–æ —Ä–∞—Å—á–µ—Ç –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤)"""
 if 'close' not in data.columns:
 raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")

 self.short_ma = data['close'].rolling(window=self.short_window).mean()
 self.long_ma = data['close'].rolling(window=self.long_window).mean()
 self.is_trained = True

 def predict(self, data: pd.DataFrame) -> pd.Series:
 """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Ç–æ—Ä–≥–æ–≤—ã—Ö —Å–∏–≥–Ω–∞–ª–æ–≤"""
 if not self.is_trained:
 raise ValueError("–°—Ç—Ä–∞—Ç–µ–≥–∏—è not –æ–±—É—á–µ–Ω–∞. –í—ã–∑–æ–≤–∏—Ç–µ train() —Å–Ω–∞—á–∞–ª–∞.")

 if 'close' not in data.columns:
 raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")

 # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º —Å–∫–æ–ª—å–∑—è—â–∏–µ —Å—Ä–µ–¥–Ω–∏–µ for –Ω–æ–≤—ã—Ö –¥–∞–Ω–Ω—ã—Ö
 short_ma = data['close'].rolling(window=self.short_window).mean()
 long_ma = data['close'].rolling(window=self.long_window).mean()

 # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Å–∏–≥–Ω–∞–ª—ã: 1 = –ø–æ–∫—É–ø–∫–∞, -1 = –ø—Ä–æ–¥–∞–∂–∞, 0 = —É–¥–µ—Ä–∂–∞–Ω–∏–µ
 signals = pd.Series(0, index=data.index)

 # –°–∏–≥–Ω–∞–ª –ø–æ–∫—É–ø–∫–∏: –∫–æ—Ä–æ—Ç–∫–∞—è MA –ø–µ—Ä–µ—Å–µ–∫–∞–µ—Ç –¥–ª–∏–Ω–Ω—É—é MA —Å–Ω–∏–∑—É –≤–≤–µ—Ä—Ö
 buy_signal = (short_ma > long_ma) & (short_ma.shift(1) <= long_ma.shift(1))

 # –°–∏–≥–Ω–∞–ª –ø—Ä–æ–¥–∞–∂–∏: –∫–æ—Ä–æ—Ç–∫–∞—è MA –ø–µ—Ä–µ—Å–µ–∫–∞–µ—Ç –¥–ª–∏–Ω–Ω—É—é MA —Å–≤–µ—Ä—Ö—É –≤–Ω–∏–∑
 sell_signal = (short_ma < long_ma) & (short_ma.shift(1) >= long_ma.shift(1))

 signals[buy_signal] = 1
 signals[sell_signal] = -1

 return signals

 def get_name(self) -> str:
 return f"SMA_{self.short_window}_{self.long_window}"

class Backtester:
 """–ö–ª–∞—Å—Å for –ø—Ä–æ–≤–µ–¥–µ–Ω–∏—è –±—ç–∫—Ç–µ—Å—Ç–∏–Ω–≥–∞ —Ç–æ—Ä–≥–æ–≤—ã—Ö —Å—Ç—Ä–∞—Ç–µ–≥–∏–π"""

 def __init__(self, initial_capital: float = 100000.0, commission: float = 0.001):
 self.initial_capital = initial_capital
 self.commission = commission

 def run_backtest(self, data: pd.DataFrame, strategy: TradingStrategy) -> Dict[str, float]:
 """
 –ó–∞–ø—É—Å–∫ –±—ç–∫—Ç–µ—Å—Ç–∏–Ω–≥–∞ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏

 Args:
 data: –î–∞–Ω–Ω—ã–µ for —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏—è
 strategy: –û–±—É—á–µ–Ω–Ω–∞—è —Ç–æ—Ä–≥–æ–≤–∞—è —Å—Ç—Ä–∞—Ç–µ–≥–∏—è

 Returns:
 –°–ª–æ–≤–∞—Ä—å with –º–µ—Ç—Ä–∏–∫–∞–º–∏ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
 """
 if 'close' not in data.columns:
 raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")

 # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Ç–æ—Ä–≥–æ–≤—ã–µ —Å–∏–≥–Ω–∞–ª—ã
 signals = strategy.predict(data)

 # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç—å
 returns = data['close'].pct_change()

 # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º —Å—Ç—Ä–∞—Ç–µ–≥–∏—á–µ—Å–∫—É—é –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç—å
 strategy_returns = signals.shift(1) * returns # –°–¥–≤–∏–≥–∞–µ–º —Å–∏–≥–Ω–∞–ª—ã on 1 –ø–µ—Ä–∏–æ–¥

 # –£—á–∏—Ç—ã–≤–∞–µ–º –∫–æ–º–∏—Å—Å–∏—é
 position_changes = signals.diff().abs()
 strategy_returns -= position_changes * self.commission

 # –£–¥–∞–ª—è–µ–º NaN –∑–Ω–∞—á–µ–Ω–∏—è
 strategy_returns = strategy_returns.dropna()

 if len(strategy_returns) == 0:
 return {
 'total_return': 0.0,
 'sharpe_ratio': 0.0,
 'max_drawdown': 0.0,
 'win_rate': 0.0,
 'total_trades': 0
 }

 # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º –∫—É–º—É–ª—è—Ç–∏–≤–Ω—É—é –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç—å
 cumulative_returns = (1 + strategy_returns).cumprod()

 # –û—Å–Ω–æ–≤–Ω—ã–µ –º–µ—Ç—Ä–∏–∫–∏
 total_return = cumulative_returns.iloc[-1] - 1

 # Sharpe Ratio (–ø—Ä–µ–¥–ø–æ–ª–∞–≥–∞–µ–º –±–µ–∑—Ä–∏—Å–∫–æ–≤—É—é —Å—Ç–∞–≤–∫—É = 0)
 sharpe_ratio = strategy_returns.mean() / strategy_returns.std() * np.sqrt(252) if strategy_returns.std() > 0 else 0

 # –ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è –ø—Ä–æ—Å–∞–¥–∫–∞
 rolling_max = cumulative_returns.expanding().max()
 drawdowns = (cumulative_returns - rolling_max) / rolling_max
 max_drawdown = drawdowns.min()

 # –ü—Ä–æ—Ü–µ–Ω—Ç –≤—ã–∏–≥—Ä—ã—à–Ω—ã—Ö —Å–¥–µ–ª–æ–∫
 winning_trades = strategy_returns[strategy_returns > 0]
 total_trades = len(strategy_returns[strategy_returns != 0])
 win_rate = len(winning_trades) / total_trades if total_trades > 0 else 0

 return {
 'total_return': total_return,
 'sharpe_ratio': sharpe_ratio,
 'max_drawdown': max_drawdown,
 'win_rate': win_rate,
 'total_trades': total_trades,
 'strategy_returns': strategy_returns,
 'cumulative_returns': cumulative_returns
 }

class WalkForwardAnalyzer:

 'mean_win_rate': np.nanmean(win_rates),

 'total_trades': np.nansum(total_trades),

 # –ö–æ–Ω—Å–∏—Å—Ç–µ–Ω—Ç–Ω–æ—Å—Ç—å
 'positive_periods': np.sum(returns > 0),

 'consistency': np.sum(returns > 0) / len(returns) if len(returns) > 0 else 0,

 # –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ –º–µ—Ç—Ä–∏–∫–∏
 'coefficient_of_variation': np.nanstd(returns) / np.abs(np.nanmean(returns)) if np.nanmean(returns) != 0 else np.inf,
 'skewness': self._calculate_skewness(returns),
 'kurtosis': self._calculate_kurtosis(returns),

 # –î–µ—Ç–∞–ª—å–Ω—ã–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã
 'period_returns': returns.tolist(),
 'period_sharpe_ratios': sharpe_ratios.tolist(),
 'period_drawdowns': max_drawdowns.tolist()
 }

 return analysis

 def _calculate_skewness(self, data: np.ndarray) -> float:
 """–†–∞—Å—á–µ—Ç –∞—Å–∏–º–º–µ—Ç—Ä–∏–∏ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è"""
 data_clean = data[~np.isnan(data)]
 if len(data_clean) < 3:
 return 0.0

 mean = np.mean(data_clean)
 std = np.std(data_clean)
 if std == 0:
 return 0.0

 return np.mean(((data_clean - mean) / std) ** 3)

 def _calculate_kurtosis(self, data: np.ndarray) -> float:
 """–†–∞—Å—á–µ—Ç —ç–∫—Å—Ü–µ—Å—Å–∞ —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è"""
 data_clean = data[~np.isnan(data)]
 if len(data_clean) < 4:
 return 0.0

 mean = np.mean(data_clean)
 std = np.std(data_clean)
 if std == 0:
 return 0.0

 return np.mean(((data_clean - mean) / std) ** 4) - 3
```

def compare_strategies(results: Dict[str, Dict[str, Any]]) -> List[Tuple[str, Dict[str, float]]]:
 """
 –°—Ä–∞–≤–Ω–µ–Ω–∏–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ –º–Ω–æ–∂–µ—Å—Ç–≤–µ–Ω–Ω—ã—Ö —Å—Ç—Ä–∞—Ç–µ–≥–∏–π.

 –≠—Ç–∞ function —Å—Ä–∞–≤–Ω–∏–≤–∞–µ—Ç —Ä–µ–∑—É–ª—å—Ç–∞—Ç—ã Walk-Forward –∞–Ω–∞–ª–∏–∑–∞
 for —Ä–∞–∑–Ω—ã—Ö —Å—Ç—Ä–∞—Ç–µ–≥–∏–π and —Ä–∞–Ω–∂–∏—Ä—É–µ—Ç –∏—Ö on –∫–ª—é—á–µ–≤—ã–º –º–µ—Ç—Ä–∏–∫–∞–º.

 Args:
 results: –°–ª–æ–≤–∞—Ä—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ –∞–Ω–∞–ª–∏–∑–∞ —Å—Ç—Ä–∞—Ç–µ–≥–∏–π

 Returns:
 –û—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã–π —Å–ø–∏—Å–æ–∫ —Å—Ç—Ä–∞—Ç–µ–≥–∏–π with –º–µ—Ç—Ä–∏–∫–∞–º–∏
 """

 comparison = {}

 for strategy_name, analysis in results.items():
 if analysis is None:
 continue

 comparison[strategy_name] = {
 'mean_return': analysis['mean_return'],
 'consistency': analysis['consistency'],
 'mean_sharpe': analysis['mean_sharpe'],
 'worst_drawdown': analysis['worst_drawdown'],
 'coefficient_of_variation': analysis['coefficient_of_variation'],
 'total_periods': analysis['total_periods']
 }

 # –°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ on —Å—Ä–µ–¥–Ω–µ–π –¥–æ—Ö–æ–¥–Ω–æ—Å—Ç–∏ (on —É–±—ã–≤–∞–Ω–∏—é)
 sorted_strategies = sorted(
 comparison.items(),
 key=lambda x: x[1]['mean_return'],
 reverse=True
 )

 return sorted_strategies

class RSIStrategy(TradingStrategy):
 """–°—Ç—Ä–∞—Ç–µ–≥–∏—è on basis RSI –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–∞"""

 def __init__(self, rsi_period: int = 14, oversold: float = 30, overbought: float = 70):
 self.rsi_period = rsi_period
 self.oversold = oversold
 self.overbought = overbought
 self.is_trained = False

 def train(self, data: pd.DataFrame) -> None:
 """–û–±—É—á–µ–Ω–∏–µ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏"""
 if 'close' not in data.columns:
 raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")
 self.is_trained = True

 def predict(self, data: pd.DataFrame) -> pd.Series:
 """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Å–∏–≥–Ω–∞–ª–æ–≤ on basis RSI"""
 if not self.is_trained:
 raise ValueError("–°—Ç—Ä–∞—Ç–µ–≥–∏—è not –æ–±—É—á–µ–Ω–∞")

 if 'close' not in data.columns:
 raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")

 # –†–∞—Å—Å—á–∏—Ç—ã–≤–∞–µ–º RSI
 delta = data['close'].diff()
 gain = (delta.where(delta > 0, 0)).rolling(window=self.rsi_period).mean()
 loss = (-delta.where(delta < 0, 0)).rolling(window=self.rsi_period).mean()
 rs = gain / loss
 rsi = 100 - (100 / (1 + rs))

 # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Å–∏–≥–Ω–∞–ª—ã
 signals = pd.Series(0, index=data.index)

 # –°–∏–≥–Ω–∞–ª –ø–æ–∫—É–ø–∫–∏: RSI –≤—ã—Ö–æ–¥–∏—Ç –∏–∑ –∑–æ–Ω—ã –ø–µ—Ä–µ–ø—Ä–æ–¥–∞–Ω–Ω–æ—Å—Ç–∏
 buy_signal = (rsi > self.oversold) & (rsi.shift(1) <= self.oversold)

 # –°–∏–≥–Ω–∞–ª –ø—Ä–æ–¥–∞–∂–∏: RSI –≤—ã—Ö–æ–¥–∏—Ç –∏–∑ –∑–æ–Ω—ã –ø–µ—Ä–µ–∫—É–ø–ª–µ–Ω–Ω–æ—Å—Ç–∏
 sell_signal = (rsi < self.overbought) & (rsi.shift(1) >= self.overbought)

 signals[buy_signal] = 1
 signals[sell_signal] = -1

 return signals

 def get_name(self) -> str:
 return f"RSI_{self.rsi_period}_{self.oversold}_{self.overbought}"

class BollingerBandsStrategy(TradingStrategy):
 """–°—Ç—Ä–∞—Ç–µ–≥–∏—è on basis –ø–æ–ª–æ—Å –ë–æ–ª–ª–∏–Ω–¥–∂–µ—Ä–∞"""

 def __init__(self, period: int = 20, std_dev: float = 2.0):
 self.period = period
 self.std_dev = std_dev
 self.is_trained = False

 def train(self, data: pd.DataFrame) -> None:
 """–û–±—É—á–µ–Ω–∏–µ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏"""
 if 'close' not in data.columns:
 raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")
 self.is_trained = True

 def predict(self, data: pd.DataFrame) -> pd.Series:
 """–ì–µ–Ω–µ—Ä–∞—Ü–∏—è —Å–∏–≥–Ω–∞–ª–æ–≤ on basis –ø–æ–ª–æ—Å –ë–æ–ª–ª–∏–Ω–¥–∂–µ—Ä–∞"""
 if not self.is_trained:
 raise ValueError("–°—Ç—Ä–∞—Ç–µ–≥–∏—è not –æ–±—É—á–µ–Ω–∞")

 if 'close' not in data.columns:
 raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã —Å–æ–¥–µ—Ä–∂–∞—Ç—å –∫–æ–ª–æ–Ω–∫—É 'close'")

 sma = data['close'].rolling(window=self.period).mean()
 std = data['close'].rolling(window=self.period).std()
 upper_band = sma + (std * self.std_dev)
 lower_band = sma - (std * self.std_dev)

 # –ì–µ–Ω–µ—Ä–∏—Ä—É–µ–º —Å–∏–≥–Ω–∞–ª—ã
 signals = pd.Series(0, index=data.index)

 # –°–∏–≥–Ω–∞–ª –ø–æ–∫—É–ø–∫–∏: —Ü–µ–Ω–∞ –∫–∞—Å–∞–µ—Ç—Å—è –Ω–∏–∂–Ω–µ–π –ø–æ–ª–æ—Å—ã
 buy_signal = (data['close'] <= lower_band) & (data['close'].shift(1) > lower_band.shift(1))

 # –°–∏–≥–Ω–∞–ª –ø—Ä–æ–¥–∞–∂–∏: —Ü–µ–Ω–∞ –∫–∞—Å–∞–µ—Ç—Å—è –≤–µ—Ä—Ö–Ω–µ–π –ø–æ–ª–æ—Å—ã
 sell_signal = (data['close'] >= upper_band) & (data['close'].shift(1) < upper_band.shift(1))

 signals[buy_signal] = 1
 signals[sell_signal] = -1

 return signals

 def get_name(self) -> str:
 return f"BB_{self.period}_{self.std_dev}"
```

def rolling_walk_forward(data, strategy, window_size=252, test_size=63):
 """Rolling Window Walk-Forward"""

 results = []
 start_idx = 0

 while start_idx + window_size + test_size <= len(data):
 # –û–±—É—á–∞—é—â–∏–π –ø–µ—Ä–∏–æ–¥ (—Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –æ–∫–Ω–æ)

 strategy.train(train_data)
 backtester = Backtester()
 metrics = backtester.run_backtest(test_data, strategy)

 start_idx += test_size

 return results

def expanding_walk_forward(data, strategy, min_train=126, test_size=63):
 """Expanding Window Walk-Forward"""

 results = []
 start_idx = 0
 train_size = min_train

 while start_idx + train_size + test_size <= len(data):
 # –û–±—É—á–∞—é—â–∏–π –ø–µ—Ä–∏–æ–¥ (—Ä–∞—Å—à–∏—Ä—è—é—â–µ–µ—Å—è –æ–∫–Ω–æ)

 # –¢–µ—Å—Ç–æ–≤—ã–π –ø–µ—Ä–∏–æ–¥
 test_data = data.iloc[start_idx + train_size:start_idx + train_size + test_size]

 # –û–±—É—á–µ–Ω–∏–µ and —Ç–µ—Å—Ç–∏—Ä–æ–≤–∞–Ω–∏–µ
 strategy.train(train_data)
 backtester = Backtester()
 metrics = backtester.run_backtest(test_data, strategy)

 results.append(metrics)
 start_idx += test_size
 train_size += test_size # –†–∞—Å—à–∏—Ä—è–µ–º –æ–∫–Ω–æ

 return results

def analyze_degradation(results):

 'second_half_mean': second_half_mean,
