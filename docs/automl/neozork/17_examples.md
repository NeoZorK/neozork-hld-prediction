### –î–µ—Ç–∞–ª—å–Ω–æ–µ description —Ä–µ–∞–ª–∏–∑–∞—Ü–∏–∏

**–ö–ª—é—á–µ–≤—ã–µ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç—ã —Å–∏—Å—Ç–µ–º—ã:**
1. **–ò–∑–≤–ª–µ—á–µ–Ω–∏–µ –ø—Ä–∏–∑–Ω–∞–∫–æ–≤:** create —Ç–µ—Ö–Ω–∏—á–µ—Å–∫–∏—Ö indicators (RSI, MACD, —Å–∫–æ–ª—å–∑—è—â–∏–µ —Å—Ä–µ–¥–Ω–∏–µ)
2. **–¶–µ–ª–µ–≤–∞—è –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è:** –ö–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è –¥–≤–∏–∂–µ–Ω–∏—è —Ü–µ–Ω—ã on 3 –∫–ª–∞—Å—Å–∞ (–≤–Ω–∏–∑, —É–¥–µ—Ä–∂–∞–Ω–∏–µ, –≤–≤–µ—Ä—Ö)

4. **–ë—ç–∫—Ç–µ—Å—Ç–∏–Ω–≥:** check —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏ on –∏—Å—Ç–æ—Ä–∏—á–µ—Å–∫–∏—Ö –¥–∞–Ω–Ω—ã—Ö

**–ü–æ—á–µ–º—É Random Forest –ø–æ–¥—Ö–æ–¥–∏—Ç for WAVE2:**
- **–£—Å—Ç–æ–π—á–∏–≤–æ—Å—Ç—å –∫ –ø–µ—Ä–µ–æ–±—É—á–µ–Ω–∏—é:** –í–∞–∂–Ω–æ for —Ñ–∏–Ω–∞–Ω—Å–æ–≤—ã—Ö –¥–∞–Ω–Ω—ã—Ö
- **–û–±—Ä–∞–±–æ—Ç–∫–∞ –Ω–µ–ª–∏–Ω–µ–π–Ω–æ—Å—Ç–∏:** –í–æ–ª–Ω–æ–≤—ã–µ –ø–∞—Ç—Ç–µ—Ä–Ω—ã —á–∞—Å—Ç–æ –Ω–µ–ª–∏–Ω–µ–π–Ω—ã
- **–ò–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∏—Ä—É–µ–º–æ—Å—Ç—å:** –ú–æ–∂–Ω–æ –ø–æ–Ω—è—Ç—å –≤–∞–∂–Ω–æ—Å—Ç—å –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
- **–ë—ã—Å—Ç—Ä–æ—Ç–∞ –æ–±—É—á–µ–Ω–∏—è:** –ö—Ä–∏—Ç–∏—á–Ω–æ for —á–∞—Å—Ç–æ–≥–æ –ø–µ—Ä–µ–æ–±—É—á–µ–Ω–∏—è

```python
import pandas as pd
import numpy as np
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score, classification_report
import yfinance as yf
from datetime import datetime, timedelta
import warnings

# Installation –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–µ–π (–≤—ã–ø–æ–ª–Ω–∏—Ç—å –ø–µ—Ä–µ–¥ Launch–æ–º):

class SimpleWave2System:
 """–ü—Ä–æ—Å—Ç–∞—è —Å–∏—Å—Ç–µ–º–∞ on basis WAVE2"""

 def __init__(self, symbol='BTC-USD', timeframe='1h'):
 self.symbol = symbol
 self.timeframe = timeframe
 self.model = RandomForestClassifier(n_estimators=100, random_state=42)
 self.features = []
 self.target = []

 def load_data(self, period='1y'):
 """
 –ó–∞–≥—Ä—É–∑–∫–∞ —Ä—ã–Ω–æ—á–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö

 Args:
 period (str): –ü–µ—Ä–∏–æ–¥ –¥–∞–Ω–Ω—ã—Ö ('1y', '2y', '5y', 'max')

 Returns:
 pd.DataFrame: –î–∞–Ω–Ω—ã–µ OHLCV with –≤—Ä–µ–º–µ–Ω–Ω—ã–º–∏ –º–µ—Ç–∫–∞–º–∏

 –¢–µ–æ—Ä–∏—è: –ó–∞–≥—Ä—É–∑–∫–∞ –∫–∞—á–µ—Å—Ç–≤–µ–Ω–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö –∫—Ä–∏—Ç–∏—á–µ—Å–∫–∏ –≤–∞–∂–Ω–∞ for –æ–±—É—á–µ–Ω–∏—è ML-–º–æ–¥–µ–ª–∏.
 –ò—Å–ø–æ–ª—å–∑—É–µ–º yfinance for –ø–æ–ª—É—á–µ–Ω–∏—è –∏—Å—Ç–æ—Ä–∏—á–µ—Å–∫–∏—Ö –¥–∞–Ω–Ω—ã—Ö with –±–∏—Ä–∂–∏.
 """
 try:
 ticker = yf.Ticker(self.symbol)
 data = ticker.history(period=period, interval=self.timeframe)

 if data.empty:
 raise ValueError(f"No data available for {self.symbol}")

 # check –∫–∞—á–µ—Å—Ç–≤–∞ –¥–∞–Ω–Ω—ã—Ö
 if len(data) < 100:
 raise ValueError(f"Insufficient data: {len(data)} rows")

 print(f"Loaded {len(data)} data points for {self.symbol}")
 return data

 except Exception as e:
 print(f"Error loading data: {e}")
 return None

 def create_wave2_features(self, data):
 """
 create –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ for WAVE2 —Å–∏—Å—Ç–µ–º—ã

 Args:
 data (pd.DataFrame): OHLCV –¥–∞–Ω–Ω—ã–µ

 Returns:
 pd.DataFrame: –ü—Ä–∏–∑–Ω–∞–∫–∏ for ML-–º–æ–¥–µ–ª–∏

 –¢–µ–æ—Ä–∏—è: WAVE2 –ø—Ä–∏–∑–Ω–∞–∫–∏ –æ—Å–Ω–æ–≤–∞–Ω—ã on –≤–æ–ª–Ω–æ–≤–æ–º –∞–Ω–∞–ª–∏–∑–µ –≠–ª–ª–∏–æ—Ç—Ç–∞:
 1. –ë–∞–∑–æ–≤—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏ - —Ü–µ–Ω–∞ and –æ–±—ä–µ–º
 2. –¢–µ—Ö–Ω–∏—á–µ—Å–∫–∏–µ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä—ã - RSI, MACD, —Å–∫–æ–ª—å–∑—è—â–∏–µ —Å—Ä–µ–¥–Ω–∏–µ
 3. –ú–æ–º–µ–Ω—Ç—É–º –ø—Ä–∏–∑–Ω–∞–∫–∏ - –∏–∑–º–µ–Ω–µ–Ω–∏—è —Ü–µ–Ω—ã and –æ–±—ä–µ–º–∞
 4. –õ–∞–≥–æ–≤—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏ - –∏—Å—Ç–æ—Ä–∏—á–µ—Å–∫–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è for —É—á–µ—Ç–∞ —Ç—Ä–µ–Ω–¥–æ–≤
 5. –í–æ–ª–∞—Ç–∏–ª—å–Ω–æ—Å—Ç—å - for –æ—Ü–µ–Ω–∫–∏ —Ä–∏—Å–∫–∞
 """
 features = pd.DataFrame(index=data.index)

 # –ë–∞–∑–æ–≤—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏
 features['close'] = data['Close']
 features['high'] = data['High']
 features['low'] = data['Low']
 features['volume'] = data['Volume']

 # –¶–µ–Ω–æ–≤—ã–µ –æ—Ç–Ω–æ—à–µ–Ω–∏—è (–≤–∞–∂–Ω–æ for –≤–æ–ª–Ω–æ–≤–æ–≥–æ –∞–Ω–∞–ª–∏–∑–∞)
 features['hl_ratio'] = data['High'] / data['Low']
 features['co_ratio'] = data['Close'] / data['Open']
 features['price_range'] = (data['High'] - data['Low']) / data['Close']

 # –¢–µ—Ö–Ω–∏—á–µ—Å–∫–∏–µ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä—ã
 features['sma_5'] = data['Close'].rolling(5).mean()
 features['sma_20'] = data['Close'].rolling(20).mean()
 features['sma_50'] = data['Close'].rolling(50).mean()
 features['rsi'] = self._calculate_rsi(data['Close'])
 features['macd'] = self._calculate_macd(data['Close'])

 # WAVE2-–ø–æ–¥–æ–±–Ω—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏ (–æ—Å–Ω–æ–≤–∞ –≤–æ–ª–Ω–æ–≤–æ–≥–æ –∞–Ω–∞–ª–∏–∑–∞)
 features['price_momentum_1'] = data['Close'].pct_change(1)
 features['price_momentum_5'] = data['Close'].pct_change(5)
 features['price_momentum_10'] = data['Close'].pct_change(10)
 features['volume_momentum'] = data['Volume'].pct_change(5)
 features['volatility_5'] = data['Close'].rolling(5).std()
 features['volatility_20'] = data['Close'].rolling(20).std()

 # –í–æ–ª–Ω–æ–≤—ã–µ –ø–∞—Ç—Ç–µ—Ä–Ω—ã (—É–ø—Ä–æ—â–µ–Ω–Ω—ã–µ)
 features['wave_up'] = ((data['Close'] > data['Close'].shift(1)) &
 (data['Close'].shift(1) > data['Close'].shift(2))).astype(int)
 features['wave_down'] = ((data['Close'] < data['Close'].shift(1)) &
 (data['Close'].shift(1) < data['Close'].shift(2))).astype(int)

 # –õ–∞–≥–æ–≤—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏ (–∫—Ä–∏—Ç–∏—á–Ω–æ for —É—á–µ—Ç–∞ —Ç—Ä–µ–Ω–¥–æ–≤)
 for lag in [1, 2, 3, 5, 10, 20]:
 features[f'close_lag_{lag}'] = data['Close'].shift(lag)
 features[f'volume_lag_{lag}'] = data['Volume'].shift(lag)
 features[f'high_lag_{lag}'] = data['High'].shift(lag)
 features[f'low_lag_{lag}'] = data['Low'].shift(lag)

 # –°–∫–æ–ª—å–∑—è—â–∏–µ —Å—Ä–µ–¥–Ω–∏–µ —Ä–∞–∑–Ω—ã—Ö periods
 for window in [3, 5, 10, 20, 50]:
 features[f'sma_{window}'] = data['Close'].rolling(window).mean()
 features[f'std_{window}'] = data['Close'].rolling(window).std()
 features[f'min_{window}'] = data['Low'].rolling(window).min()
 features[f'max_{window}'] = data['High'].rolling(window).max()

 # –û—Ç–Ω–æ—à–µ–Ω–∏—è –∫ —Å–∫–æ–ª—å–∑—è—â–∏–º —Å—Ä–µ–¥–Ω–∏–º (–≤–∞–∂–Ω–æ for –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è —Ç—Ä–µ–Ω–¥–∞)
 features['close_vs_sma_20'] = data['Close'] / features['sma_20']
 features['close_vs_sma_50'] = data['Close'] / features['sma_50']
 features['sma_20_vs_sma_50'] = features['sma_20'] / features['sma_50']

 return features

 def _calculate_rsi(self, prices, window=14):
 """–†–∞—Å—á–µ—Ç RSI"""
 delta = prices.diff()
 gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
 loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
 rs = gain / loss
 rsi = 100 - (100 / (1 + rs))
 return rsi

 def _calculate_macd(self, prices, fast=12, slow=26, signal=9):
 """–†–∞—Å—á–µ—Ç MACD"""
 ema_fast = prices.ewm(span=fast).mean()
 ema_slow = prices.ewm(span=slow).mean()
 macd = ema_fast - ema_slow
 signal_line = macd.ewm(span=signal).mean()
 return macd - signal_line

 def create_target(self, data, horizon=1):
 """create —Ü–µ–ª–µ–≤–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π"""
 future_price = data['Close'].shift(-horizon)
 current_price = data['Close']

 # –ü—Ä–æ—Ü–µ–Ω—Ç–Ω–æ–µ –∏–∑–º–µ–Ω–µ–Ω–∏–µ
 price_change = (future_price - current_price) / current_price

 # –ö–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è
 target = pd.cut(
 price_change,
 bins=[-np.inf, -0.01, 0.01, np.inf],
 labels=[0, 1, 2], # 0=down, 1=hold, 2=up
 include_lowest=True
 )

 return target.astype(int)

 def train_model(self, data):
 """–û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–∏"""
 # create –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
 features = self.create_wave2_features(data)

 # create —Ü–µ–ª–µ–≤–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π
 target = self.create_target(data)

 # remove NaN
 valid_indices = ~(features.isna().any(axis=1) | target.isna())
 features_clean = features[valid_indices]
 target_clean = target[valid_indices]

 # –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ on train/test
 X_train, X_test, y_train, y_test = train_test_split(
 features_clean, target_clean, test_size=0.2, random_state=42
 )

 # –û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–∏
 self.model.fit(X_train, y_train)

 # Prediction
 y_pred = self.model.predict(X_test)

 # –û—Ü–µ–Ω–∫–∞
```

SCHR (Support, Channel, High, Resistance) Levels - —ç—Ç–æ –ø—Ä–æ–¥–≤–∏–Ω—É—Ç–∞—è —Å–∏—Å—Ç–µ–º–∞ –∞–Ω–∞–ª–∏–∑–∞ —É—Ä–æ–≤–Ω–µ–π –ø–æ–¥–¥–µ—Ä–∂–∫–∏ and —Å–æ–ø—Ä–æ—Ç–∏–≤–ª–µ–Ω–∏—è, –æ—Å–Ω–æ–≤–∞–Ω–Ω–∞—è on –º–∞—à–∏–Ω–Ω–æ–º –æ–±—É—á–µ–Ω–∏–∏. –°–∏—Å—Ç–µ–º–∞ –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç –∫–ª—é—á–µ–≤—ã–µ —É—Ä–æ–≤–Ω–∏, –≥–¥–µ —Ü–µ–Ω–∞ with –≤—ã—Å–æ–∫–æ–π –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å—é –º–æ–∂–µ—Ç —Ä–∞–∑–≤–µ—Ä–Ω—É—Ç—å—Å—è.

**–ö–ª—é—á–µ–≤—ã–µ –ø—Ä–∏–Ω—Ü–∏–ø—ã SCHR Levels:**
1. **–ü–æ–¥–¥–µ—Ä–∂–∫–∞ (Support):** –£—Ä–æ–≤–Ω–∏, –≥–¥–µ —Ü–µ–Ω–∞ –Ω–∞—Ö–æ–¥–∏—Ç –ø–æ–¥–¥–µ—Ä–∂–∫—É and –æ—Ç—Å–∫–∞–∫–∏–≤–∞–µ—Ç –≤–≤–µ—Ä—Ö
2. **–°–æ–ø—Ä–æ—Ç–∏–≤–ª–µ–Ω–∏–µ (Resistance):** –£—Ä–æ–≤–Ω–∏, –≥–¥–µ —Ü–µ–Ω–∞ –≤—Å—Ç—Ä–µ—á–∞–µ—Ç —Å–æ–ø—Ä–æ—Ç–∏–≤–ª–µ–Ω–∏–µ and –æ—Ç—Å–∫–∞–∫–∏–≤–∞–µ—Ç –≤–Ω–∏–∑
3. **–ö–∞–Ω–∞–ª—ã (Channels):** –î–∏–∞–ø–∞–∑–æ–Ω—ã —Ü–µ–Ω, in –∫–æ—Ç–æ—Ä—ã—Ö –¥–≤–∏–∂–µ—Ç—Å—è –∞–∫—Ç–∏–≤
4. **–î–∞–≤–ª–µ–Ω–∏–µ (Pressure):** –°–∏–ª–∞ –ø–æ–∫—É–ø–∞—Ç–µ–ª–µ–π/–ø—Ä–æ–¥–∞–≤—Ü–æ–≤ on –∫–∞–∂–¥–æ–º —É—Ä–æ–≤–Ω–µ

**–ü–æ—á–µ–º—É SCHR Levels —ç—Ñ—Ñ–µ–∫—Ç–∏–≤–Ω—ã:**
- **–í—ã—Å–æ–∫–∞—è —Ç–æ—á–Ω–æ—Å—Ç—å:** ML-–º–æ–¥–µ–ª—å —É—á–∏—Ç—Å—è on –∏—Å—Ç–æ—Ä–∏—á–µ—Å–∫–∏—Ö –ø–∞—Ç—Ç–µ—Ä–Ω–∞—Ö
- **–ê–¥–∞–ø—Ç–∏–≤–Ω–æ—Å—Ç—å:** –°–∏—Å—Ç–µ–º–∞ –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ –∞–¥–∞–ø—Ç–∏—Ä—É–µ—Ç—Å—è –∫ –∏–∑–º–µ–Ω–µ–Ω–∏—è–º —Ä—ã–Ω–∫–∞
- **–†–æ–±–∞—Å—Ç–Ω–æ—Å—Ç—å:** –£—Å—Ç–æ–π—á–∏–≤–∞ –∫ —Ä—ã–Ω–æ—á–Ω–æ–º—É —à—É–º—É and –ª–æ–∂–Ω—ã–º —Å–∏–≥–Ω–∞–ª–∞–º
- **–ú–∞—Å—à—Ç–∞–±–∏—Ä—É–µ–º–æ—Å—Ç—å:** –†–∞–±–æ—Ç–∞–µ—Ç on —Ä–∞–∑–Ω—ã—Ö Timeframe–∞—Ö and –∞–∫—Ç–∏–≤–∞—Ö

**–ê–Ω—Å–∞–º–±–ª—å –º–æ–¥–µ–ª–µ–π:**

- **XGBoost:** –ë—ã—Å—Ç—Ä–∞—è and —Ç–æ—á–Ω–∞—è –º–æ–¥–µ–ª—å for –±–∞–∑–æ–≤—ã—Ö predictions
- **Gradient Boosting:** –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω–∞—è –º–æ–¥–µ–ª—å for —É–ª—É—á—à–µ–Ω–∏—è —Ç–æ—á–Ω–æ—Å—Ç–∏
- **–í–∑–≤–µ—à–µ–Ω–Ω–æ–µ –≥–æ–ª–æ—Å–æ–≤–∞–Ω–∏–µ:** –û–±—ä–µ–¥–∏–Ω—è–µ–º –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏—è with –æ–ø—Ç–∏–º–∞–ª—å–Ω—ã–º–∏ –≤–µ—Å–∞–º–∏

```python
 def load_data(self, period='2y'):

 Returns:

 def create_schr_features(self, data):
 """
 create —Ä–∞—Å—à–∏—Ä–µ–Ω–Ω—ã—Ö –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ for SCHR Levels —Å–∏—Å—Ç–µ–º—ã

 Args:
 data (pd.DataFrame): OHLCV –¥–∞–Ω–Ω—ã–µ

 Returns:
 pd.DataFrame: –ü—Ä–∏–∑–Ω–∞–∫–∏ for ML-–º–æ–¥–µ–ª–∏

 –¢–µ–æ—Ä–∏—è: SCHR –ø—Ä–∏–∑–Ω–∞–∫–∏ –æ—Å–Ω–æ–≤–∞–Ω—ã on –∞–Ω–∞–ª–∏–∑–µ —É—Ä–æ–≤–Ω–µ–π –ø–æ–¥–¥–µ—Ä–∂–∫–∏ and —Å–æ–ø—Ä–æ—Ç–∏–≤–ª–µ–Ω–∏—è:
 1. –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω—ã–µ —É—Ä–æ–≤–Ω–∏ - ML-–º–æ–¥–µ–ª—å –ø—Ä–µ–¥—Å–∫–∞–∑—ã–≤–∞–µ—Ç –∫–ª—é—á–µ–≤—ã–µ —É—Ä–æ–≤–Ω–∏
 2. –î–∞–≤–ª–µ–Ω–∏–µ - —Å–∏–ª–∞ –ø–æ–∫—É–ø–∞—Ç–µ–ª–µ–π/–ø—Ä–æ–¥–∞–≤—Ü–æ–≤ on –∫–∞–∂–¥–æ–º —É—Ä–æ–≤–Ω–µ
 3. –ü–æ–∑–∏—Ü–∏–æ–Ω–∏—Ä–æ–≤–∞–Ω–∏–µ - –≥–¥–µ –Ω–∞—Ö–æ–¥–∏—Ç—Å—è —Ü–µ–Ω–∞ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —É—Ä–æ–≤–Ω–µ–π
 4. –ú–æ–º–µ–Ω—Ç—É–º - –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–µ and —Å–∏–ª–∞ –¥–≤–∏–∂–µ–Ω–∏—è –∫ —É—Ä–æ–≤–Ω—è–º
 """
 features = pd.DataFrame(index=data.index)

 # –ë–∞–∑–æ–≤—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏
 features['close'] = data['Close']
 features['high'] = data['High']
 features['low'] = data['Low']
 features['open'] = data['Open']
 features['volume'] = data['Volume']

 # –¶–µ–Ω–æ–≤—ã–µ –æ—Ç–Ω–æ—à–µ–Ω–∏—è
 features['hl_ratio'] = data['High'] / data['Low']
 features['co_ratio'] = data['Close'] / data['Open']
 features['price_range'] = (data['High'] - data['Low']) / data['Close']
 features['body_size'] = abs(data['Close'] - data['Open']) / data['Close']

 # SCHR Levels –ø—Ä–∏–∑–Ω–∞–∫–∏ (–æ—Å–Ω–æ–≤–∞ —Å–∏—Å—Ç–µ–º—ã)
 features['predicted_high'] = self._calculate_predicted_high(data)
 features['predicted_low'] = self._calculate_predicted_low(data)
 features['pressure'] = self._calculate_pressure(data)
 features['pressure_vector'] = self._calculate_pressure_vector(data)

 # –†–∞—Å—Å—Ç–æ—è–Ω–∏—è to —É—Ä–æ–≤–Ω–µ–π (–∫—Ä–∏—Ç–∏—á–Ω–æ for SCHR)
 features['distance_to_high'] = (features['predicted_high'] - features['close']) / features['close']
 features['distance_to_low'] = (features['close'] - features['predicted_low']) / features['close']
 features['level_range'] = (features['predicted_high'] - features['predicted_low']) / features['close']

 # –ü–æ–∑–∏—Ü–∏—è –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ —É—Ä–æ–≤–Ω–µ–π
 level_range = features['predicted_high'] - features['predicted_low']
 features['position_in_range'] = np.where(
 level_range > 0,
 (features['close'] - features['predicted_low']) / level_range,
 0.5 # –°—Ä–µ–¥–Ω—è—è –ø–æ–∑–∏—Ü–∏—è –µ—Å–ª–∏ —É—Ä–æ–≤–Ω–∏ —Ä–∞–≤–Ω—ã
 )

 # –î–∞–≤–ª–µ–Ω–∏–µ on —É—Ä–æ–≤–Ω–∏ (–Ω–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω–æ–µ)
 features['pressure_normalized'] = features['pressure'] / features['close']
 features['pressure_vector_normalized'] = features['pressure_vector'] / features['close']

 # –ò–∑–º–µ–Ω–µ–Ω–∏—è –¥–∞–≤–ª–µ–Ω–∏—è (—Ç—Ä–µ–Ω–¥ –¥–∞–≤–ª–µ–Ω–∏—è)
 features['pressure_change'] = features['pressure'].diff()
 features['pressure_vector_change'] = features['pressure_vector'].diff()
 features['pressure_acceleration'] = features['pressure_change'].diff()

 # SCHR-—Å–ø–µ—Ü–∏—Ñ–∏—á–Ω—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏
 features['near_support'] = (features['distance_to_low'] < 0.02).astype(int) # in –ø—Ä–µ–¥–µ–ª–∞—Ö 2% from –ø–æ–¥–¥–µ—Ä–∂–∫–∏
 features['near_resistance'] = (features['distance_to_high'] < 0.02).astype(int) # in –ø—Ä–µ–¥–µ–ª–∞—Ö 2% from —Å–æ–ø—Ä–æ—Ç–∏–≤–ª–µ–Ω–∏—è
 features['in_channel'] = ((features['position_in_range'] > 0.2) & (features['position_in_range'] < 0.8)).astype(int)

 # –¢–µ—Ö–Ω–∏—á–µ—Å–∫–∏–µ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä—ã
 features['rsi'] = self._calculate_rsi(data['Close'])
 features['macd'] = self._calculate_macd(data['Close'])
 features['bollinger_upper'] = self._calculate_bollinger_bands(data['Close'])[0]
 features['bollinger_lower'] = self._calculate_bollinger_bands(data['Close'])[1]
 features['bollinger_position'] = (features['close'] - features['bollinger_lower']) / (features['bollinger_upper'] - features['bollinger_lower'])

 # –î–æ–ø–æ–ª–Ω–∏—Ç–µ–ª—å–Ω—ã–µ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä—ã for SCHR
 features['atr'] = self._calculate_atr(data)
 features['stochastic'] = self._calculate_stochastic(data)
 features['williams_r'] = self._calculate_williams_r(data)

 # –õ–∞–≥–æ–≤—ã–µ –ø—Ä–∏–∑–Ω–∞–∫–∏ (–∫—Ä–∏—Ç–∏—á–Ω–æ for —É—á–µ—Ç–∞ –∏—Å—Ç–æ—Ä–∏–∏)
 for lag in [1, 2, 3, 5, 10, 20, 50]:
 features[f'close_lag_{lag}'] = data['Close'].shift(lag)
 features[f'high_lag_{lag}'] = data['High'].shift(lag)
 features[f'low_lag_{lag}'] = data['Low'].shift(lag)
 features[f'pressure_lag_{lag}'] = features['pressure'].shift(lag)
 features[f'pressure_vector_lag_{lag}'] = features['pressure_vector'].shift(lag)
 features[f'distance_to_high_lag_{lag}'] = features['distance_to_high'].shift(lag)
 features[f'distance_to_low_lag_{lag}'] = features['distance_to_low'].shift(lag)

 # –°–∫–æ–ª—å–∑—è—â–∏–µ —Å—Ä–µ–¥–Ω–∏–µ —Ä–∞–∑–Ω—ã—Ö periods
 for window in [3, 5, 10, 20, 50, 100]:
 features[f'sma_{window}'] = data['Close'].rolling(window).mean()
 features[f'std_{window}'] = data['Close'].rolling(window).std()
 features[f'min_{window}'] = data['Low'].rolling(window).min()
 features[f'max_{window}'] = data['High'].rolling(window).max()
 features[f'pressure_sma_{window}'] = features['pressure'].rolling(window).mean()
 features[f'pressure_vector_sma_{window}'] = features['pressure_vector'].rolling(window).mean()

 # –í–∑–∞–∏–º–æ–¥–µ–π—Å—Ç–≤–∏–µ –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ (–≤–∞–∂–Ω–æ for SCHR)
 features['pressure_volume_interaction'] = features['pressure'] * features['volume']
 features['level_breakout_signal'] = ((features['close'] > features['predicted_high']) |
 (features['close'] < features['predicted_low'])).astype(int)
 features['pressure_trend'] = (features['pressure_change'] > 0).astype(int)

 return features

 def _calculate_predicted_high(self, data):
 """–†–∞—Å—á–µ—Ç –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω–æ–≥–æ –º–∞–∫—Å–∏–º—É–º–∞"""
 # –£–ø—Ä–æ—â–µ–Ω–Ω–∞—è –≤–µ—Ä—Å–∏—è SCHR Levels
 high_20 = data['High'].rolling(20).max()
 high_50 = data['High'].rolling(50).max()
 return (high_20 + high_50) / 2

 def _calculate_predicted_low(self, data):
 """–†–∞—Å—á–µ—Ç –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–Ω–æ–≥–æ –º–∏–Ω–∏–º—É–º–∞"""
 # –£–ø—Ä–æ—â–µ–Ω–Ω–∞—è –≤–µ—Ä—Å–∏—è SCHR Levels
 low_20 = data['Low'].rolling(20).min()
 low_50 = data['Low'].rolling(50).min()
 return (low_20 + low_50) / 2

 def _calculate_pressure(self, data):
 """–†–∞—Å—á–µ—Ç –¥–∞–≤–ª–µ–Ω–∏—è"""
 # –£–ø—Ä–æ—â–µ–Ω–Ω–∞—è –≤–µ—Ä—Å–∏—è –¥–∞–≤–ª–µ–Ω–∏—è
 price_change = data['Close'].pct_change()
 volume = data['Volume']
 pressure = price_change * volume
 return pressure.rolling(20).mean()

 def _calculate_pressure_vector(self, data):
 """–†–∞—Å—á–µ—Ç –≤–µ–∫—Ç–æ—Ä–∞ –¥–∞–≤–ª–µ–Ω–∏—è"""
 # –£–ø—Ä–æ—â–µ–Ω–Ω–∞—è –≤–µ—Ä—Å–∏—è –≤–µ–∫—Ç–æ—Ä–∞ –¥–∞–≤–ª–µ–Ω–∏—è
 pressure = self._calculate_pressure(data)
 return pressure.diff()

 def _calculate_rsi(self, prices, window=14):
 """–†–∞—Å—á–µ—Ç RSI"""
 delta = prices.diff()
 gain = (delta.where(delta > 0, 0)).rolling(window=window).mean()
 loss = (-delta.where(delta < 0, 0)).rolling(window=window).mean()
 rs = gain / loss
 rsi = 100 - (100 / (1 + rs))
 return rsi

 def _calculate_macd(self, prices, fast=12, slow=26, signal=9):
 """–†–∞—Å—á–µ—Ç MACD"""
 ema_fast = prices.ewm(span=fast).mean()
 ema_slow = prices.ewm(span=slow).mean()
 macd = ema_fast - ema_slow
 signal_line = macd.ewm(span=signal).mean()
 return macd - signal_line

 def _calculate_bollinger_bands(self, prices, window=20, num_std=2):
 """–†–∞—Å—á–µ—Ç –ø–æ–ª–æ—Å –ë–æ–ª–ª–∏–Ω–¥–∂–µ—Ä–∞"""
 sma = prices.rolling(window).mean()
 std = prices.rolling(window).std()
 upper = sma + (std * num_std)
 lower = sma - (std * num_std)
 return upper, lower

 def _calculate_atr(self, data, window=14):
 """–†–∞—Å—á–µ—Ç Average True Range"""
 high_low = data['High'] - data['Low']
 high_close = np.abs(data['High'] - data['Close'].shift())
 low_close = np.abs(data['Low'] - data['Close'].shift())

 true_range = np.maximum(high_low, np.maximum(high_close, low_close))
 atr = true_range.rolling(window).mean()
 return atr

 def _calculate_stochastic(self, data, k_window=14, d_window=3):
 """–†–∞—Å—á–µ—Ç Stochastic Oscillator"""
 lowest_low = data['Low'].rolling(k_window).min()
 highest_high = data['High'].rolling(k_window).max()

 k_percent = 100 * ((data['Close'] - lowest_low) / (highest_high - lowest_low))
 d_percent = k_percent.rolling(d_window).mean()

 return k_percent, d_percent

 def _calculate_williams_r(self, data, window=14):
 """–†–∞—Å—á–µ—Ç Williams %R"""
 highest_high = data['High'].rolling(window).max()
 lowest_low = data['Low'].rolling(window).min()

 williams_r = -100 * ((highest_high - data['Close']) / (highest_high - lowest_low))
 return williams_r

 def create_target(self, data, horizon=1):
 """create —Ü–µ–ª–µ–≤–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π"""
 future_price = data['Close'].shift(-horizon)
 current_price = data['Close']

 # –ü—Ä–æ—Ü–µ–Ω—Ç–Ω–æ–µ –∏–∑–º–µ–Ω–µ–Ω–∏–µ
 price_change = (future_price - current_price) / current_price

 # –ö–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—è –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏—è
 target = pd.cut(
 price_change,
 bins=[-np.inf, -0.005, 0.005, np.inf],
 labels=[0, 1, 2], # 0=down, 1=hold, 2=up
 include_lowest=True
 )

 return target.astype(int)

 def train_models(self, data):
 """–û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–µ–π"""
 # create –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
 features = self.create_schr_features(data)

 # create —Ü–µ–ª–µ–≤–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π
 target = self.create_target(data)

 # remove NaN
 valid_indices = ~(features.isna().any(axis=1) | target.isna())
 features_clean = features[valid_indices]
 target_clean = target[valid_indices]

 # Time Series Split
 tscv = TimeSeriesSplit(n_splits=5)

 # –û–±—É—á–µ–Ω–∏–µ –º–æ–¥–µ–ª–µ–π
 for name, model in self.models.items():
```

```python
 'agreement': agreement,
 'confidence': confidence,
 'direction': direction
 }
```
