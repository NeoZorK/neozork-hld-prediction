```python
        'searcher': 'auto',            # –ü–æ–∏—Å–∫–æ–≤–∏–∫: 'auto', 'random', 'bayes'

    num_bag_folds=0,                   # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ñ–æ–ª–¥–æ–≤ –¥–ª—è –±—ç–≥–≥–∏–Ω–≥–∞ (0=–æ—Ç–∫–ª—é—á–∏—Ç—å)
    num_stack_levels=0,                # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —É—Ä–æ–≤–Ω–µ–π —Å—Ç–µ–∫–∏–Ω–≥–∞ (0=–æ—Ç–∫–ª—é—á–∏—Ç—å)
    auto_stack=True,                   # –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏–π —Å—Ç–µ–∫–∏–Ω–≥

    memory_limit=None,                 # –õ–∏–º–∏—Ç –ø–∞–º—è—Ç–∏ (None=–±–µ–∑ –æ–≥—Ä–∞–Ω–∏—á–µ–Ω–∏–π)
    feature_prune=True,                # –û—Ç–±–æ—Ä –ø—Ä–∏–∑–Ω–∞–∫–æ–≤

    num_bag_folds=0,                   # –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ñ–æ–ª–¥–æ–≤ –±—ç–≥–≥–∏–Ω–≥–∞
```

#### –£–ø—Ä–∞–∂–Ω–µ–Ω–∏–µ 2: –í–∞–ª–∏–¥–∞—Ü–∏—è (1 —á–∞—Å)

**–¶–µ–ª—å:** –ü—Ä–æ–≤–µ—Å—Ç–∏ –ø–æ–ª–Ω—É—é –≤–∞–ª–∏–¥–∞—Ü–∏—é –º–æ–¥–µ–ª–∏ —Å –ø–æ–Ω–∏–º–∞–Ω–∏–µ–º –≤—Å–µ—Ö –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ –≤–∞–ª–∏–¥–∞—Ü–∏–∏

```python
# –ü—Ä–æ–≤–µ–¥–∏—Ç–µ –ø–æ–ª–Ω—É—é –≤–∞–ª–∏–¥–∞—Ü–∏—é –º–æ–¥–µ–ª–∏
from sklearn.model_selection import train_test_split, cross_val_score, StratifiedKFold
from sklearn.metrics import classification_report, confusion_matrix
import numpy as np

# –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö —Å –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏
train_data, test_data = train_test_split(
    data, 
    test_size=0.2,                      # –î–æ–ª—è —Ç–µ—Å—Ç–æ–≤—ã—Ö –¥–∞–Ω–Ω—ã—Ö (20%)
    random_state=42,                    # –°–ª—É—á–∞–π–Ω–æ–µ —Å–æ—Å—Ç–æ—è–Ω–∏–µ –¥–ª—è –≤–æ—Å–ø—Ä–æ–∏–∑–≤–æ–¥–∏–º–æ—Å—Ç–∏
    stratify=data['target']             # –°—Ç—Ä–∞—Ç–∏—Ñ–∏–∫–∞—Ü–∏—è –ø–æ —Ü–µ–ª–µ–≤–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π
)

# –°–æ–∑–¥–∞–Ω–∏–µ –º–æ–¥–µ–ª–∏ —Å –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏ –≤–∞–ª–∏–¥–∞—Ü–∏–∏
predictor = TabularPredictor(
    label='target',
    problem_type='multiclass',
    eval_metric='accuracy',
    path='iris_validation_model',
    verbosity=2
)

# –û–±—É—á–µ–Ω–∏–µ —Å –ø–∞—Ä–∞–º–µ—Ç—Ä–∞–º–∏ –≤–∞–ª–∏–¥–∞—Ü–∏–∏
predictor.fit(
    train_data,
    time_limit=120,                     # –£–≤–µ–ª–∏—á–µ–Ω–Ω–æ–µ –≤—Ä–µ–º—è –¥–ª—è –ª—É—á—à–µ–≥–æ –∫–∞—á–µ—Å—Ç–≤–∞
    presets='high_quality',             # –í—ã—Å–æ–∫–æ–µ –∫–∞—á–µ—Å—Ç–≤–æ –¥–ª—è –≤–∞–ª–∏–¥–∞—Ü–∏–∏
    holdout_frac=0.2,                   # 20% –¥–∞–Ω–Ω—ã—Ö –¥–ª—è holdout –≤–∞–ª–∏–¥–∞—Ü–∏–∏
    num_bag_folds=5,                    # 5-fold –±—ç–≥–≥–∏–Ω–≥ –¥–ª—è —Å—Ç–∞–±–∏–ª—å–Ω–æ—Å—Ç–∏
    num_stack_levels=1,                 # 1 —É—Ä–æ–≤–µ–Ω—å —Å—Ç–µ–∫–∏–Ω–≥–∞
    auto_stack=True,                    # –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏–π —Å—Ç–µ–∫–∏–Ω–≥
    num_trials=20,                      # –ë–æ–ª—å—à–µ –ø–æ–ø—ã—Ç–æ–∫ –¥–ª—è hyperparameter tuning
    hyperparameter_tune_kwargs={
        'scheduler': 'local',
        'searcher': 'bayes',            # –ë–∞–π–µ—Å–æ–≤—Å–∫–∏–π –ø–æ–∏—Å–∫ –¥–ª—è –ª—É—á—à–∏—Ö —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
        'num_trials': 20,
        'search_space': 'default'
    },
    feature_prune=True,                 # –û—Ç–±–æ—Ä –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
    refit_full=True,                    # –ü–µ—Ä–µ–æ–±—É—á–µ–Ω–∏–µ –Ω–∞ –≤—Å–µ—Ö –¥–∞–Ω–Ω—ã—Ö
    set_best_to_refit_full=True
)

# –í–∞–ª–∏–¥–∞—Ü–∏—è –Ω–∞ —Ç–µ—Å—Ç–æ–≤—ã—Ö –¥–∞–Ω–Ω—ã—Ö
test_predictions = predictor.predict(test_data)
test_accuracy = predictor.evaluate(test_data, silent=True)

# –î–µ—Ç–∞–ª—å–Ω–∞—è –æ—Ü–µ–Ω–∫–∞ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
print(f"–¢–æ—á–Ω–æ—Å—Ç—å –Ω–∞ —Ç–µ—Å—Ç–µ: {test_accuracy}")

# –ö—Ä–æ—Å—Å-–≤–∞–ª–∏–¥–∞—Ü–∏—è –Ω–∞ –æ–±—É—á–∞—é—â–∏—Ö –¥–∞–Ω–Ω—ã—Ö
cv_scores = []
skf = StratifiedKFold(n_splits=5, shuffle=True, random_state=42)
for train_idx, val_idx in skf.split(train_data.drop('target', axis=1), train_data['target']):
    cv_train = train_data.iloc[train_idx]
    cv_val = train_data.iloc[val_idx]
    
    # –û–±—É—á–µ–Ω–∏–µ –Ω–∞ —Ñ–æ–ª–¥–µ
    cv_predictor = TabularPredictor(
        label='target',
        problem_type='multiclass',
        eval_metric='accuracy',
        path=f'cv_model_{len(cv_scores)}',
        verbosity=0
    )
    cv_predictor.fit(cv_train, time_limit=60, presets='medium_quality')
    
    # –û—Ü–µ–Ω–∫–∞ –Ω–∞ –≤–∞–ª–∏–¥–∞—Ü–∏–æ–Ω–Ω–æ–º —Ñ–æ–ª–¥–µ
    cv_pred = cv_predictor.predict(cv_val)
    cv_acc = cv_predictor.evaluate(cv_val, silent=True)
    cv_scores.append(cv_acc)

print(f"–°—Ä–µ–¥–Ω—è—è —Ç–æ—á–Ω–æ—Å—Ç—å CV: {np.mean(cv_scores):.4f} (+/- {np.std(cv_scores)*2:.4f})")

# –î–µ—Ç–∞–ª—å–Ω—ã–π –æ—Ç—á–µ—Ç –æ –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏
print("\n–û—Ç—á–µ—Ç –æ –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏:")
print(classification_report(test_data['target'], test_predictions))

# –ú–∞—Ç—Ä–∏—Ü–∞ –æ—à–∏–±–æ–∫
print("\n–ú–∞—Ç—Ä–∏—Ü–∞ –æ—à–∏–±–æ–∫:")
print(confusion_matrix(test_data['target'], test_predictions))

# –ê–Ω–∞–ª–∏–∑ –≤–∞–∂–Ω–æ—Å—Ç–∏ –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
feature_importance = predictor.feature_importance(test_data)
print(f"\n–í–∞–∂–Ω–æ—Å—Ç—å –ø—Ä–∏–∑–Ω–∞–∫–æ–≤:")
print(feature_importance)

# –ê–Ω–∞–ª–∏–∑ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ –ø–æ –∫–ª–∞—Å—Å–∞–º
class_names = iris.target_names
for i, class_name in enumerate(class_names):
    class_mask = test_data['target'] == i
    if np.sum(class_mask) > 0:
        class_accuracy = np.mean(test_predictions[class_mask] == test_data['target'][class_mask])
        print(f"–¢–æ—á–Ω–æ—Å—Ç—å –¥–ª—è –∫–ª–∞—Å—Å–∞ {class_name}: {class_accuracy:.4f}")

# –ê–Ω–∞–ª–∏–∑ —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç–∏ –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–π
pred_proba = predictor.predict_proba(test_data)
confidence = np.max(pred_proba, axis=1)
print(f"\n–°—Ä–µ–¥–Ω—è—è —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å –ø—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–π: {np.mean(confidence):.4f}")
print(f"–ú–∏–Ω–∏–º–∞–ª—å–Ω–∞—è —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å: {np.min(confidence):.4f}")
print(f"–ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∞—è —É–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å: {np.max(confidence):.4f}")

# –ê–Ω–∞–ª–∏–∑ –æ—à–∏–±–æ–∫
errors = test_predictions != test_data['target']
if np.sum(errors) > 0:
    print(f"\n–ê–Ω–∞–ª–∏–∑ –æ—à–∏–±–æ–∫ ({np.sum(errors)} –∏–∑ {len(test_data)}):")
    error_indices = np.where(errors)[0]
    for idx in error_indices[:5]:  # –ü–æ–∫–∞–∑—ã–≤–∞–µ–º –ø–µ—Ä–≤—ã–µ 5 –æ—à–∏–±–æ–∫
        true_class = class_names[test_data['target'].iloc[idx]]
        pred_class = class_names[test_predictions[idx]]
        confidence_error = confidence[idx]
        print(f"  –ò–Ω–¥–µ–∫—Å {idx}: –ò—Å—Ç–∏–Ω–∞={true_class}, –ü—Ä–µ–¥—Å–∫–∞–∑–∞–Ω–∏–µ={pred_class}, –£–≤–µ—Ä–µ–Ω–Ω–æ—Å—Ç—å={confidence_error:.4f}")
```

```python
# –°–æ–∑–¥–∞–π—Ç–µ –ø—Ä–æ—Å—Ç—É—é API –¥–ª—è –º–æ–¥–µ–ª–∏
from flask import Flask, request, jsonify, abort

def handle_errors(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        global error_count
        try:
            return f(*args, **kwargs)
        except Exception as e:
            error_count += 1
            logger.error(f"–û—à–∏–±–∫–∞ –≤ {f.__name__}: {str(e)}")
            logger.error(traceback.format_exc())
            return jsonify({
                'error': 'Internal Server Error',
                'message': str(e),
                'timestamp': datetime.now().isoformat()
            }), 500
    return decorated_function

def monitor_performance(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        global request_count, total_prediction_time
        start_time = time.time()
        request_count += 1
        
        result = f(*args, **kwargs)
        
        prediction_time = time.time() - start_time
        total_prediction_time += prediction_time
        
        logger.info(f"–ó–∞–ø—Ä–æ—Å {request_count}: –≤—Ä–µ–º—è –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è {prediction_time:.4f}—Å")
        return result
    return decorated_function

def validate_input_data(data):
    """–í–∞–ª–∏–¥–∞—Ü–∏—è –≤—Ö–æ–¥–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö"""
    if not isinstance(data, (list, dict)):
        raise ValueError("–î–∞–Ω–Ω—ã–µ –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å —Å–ø–∏—Å–∫–æ–º –∏–ª–∏ —Å–ª–æ–≤–∞—Ä–µ–º")
    
    if isinstance(data, dict):
        data = [data]
    
    if len(data) > Config.MAX_BATCH_SIZE:
        raise ValueError(f"–ü—Ä–µ–≤—ã—à–µ–Ω –º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —Ä–∞–∑–º–µ—Ä –±–∞—Ç—á–∞: {len(data)} > {Config.MAX_BATCH_SIZE}")
    
    validated_data = []
    for i, item in enumerate(data):
        if not isinstance(item, dict):
            raise ValueError(f"–≠–ª–µ–º–µ–Ω—Ç {i} –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å —Å–ª–æ–≤–∞—Ä–µ–º")
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ –æ–±—è–∑–∞—Ç–µ–ª—å–Ω—ã—Ö –ø—Ä–∏–∑–Ω–∞–∫–æ–≤
        for feature in Config.REQUIRED_FEATURES:
            if feature not in item:
                raise ValueError(f"–û—Ç—Å—É—Ç—Å—Ç–≤—É–µ—Ç –æ–±—è–∑–∞—Ç–µ–ª—å–Ω—ã–π –ø—Ä–∏–∑–Ω–∞–∫: {feature}")
        
        # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Ç–∏–ø–æ–≤ –∏ –¥–∏–∞–ø–∞–∑–æ–Ω–æ–≤ –∑–Ω–∞—á–µ–Ω–∏–π
        validated_item = {}
        for feature, value in item.items():
            if feature in Config.REQUIRED_FEATURES:
                try:
                    float_value = float(value)
                    min_val, max_val = Config.FEATURE_RANGES[feature]
                    if not (min_val <= float_value <= max_val):
                        raise ValueError(f"–ó–Ω–∞—á–µ–Ω–∏–µ {feature}={float_value} –≤–Ω–µ –¥–∏–∞–ø–∞–∑–æ–Ω–∞ [{min_val}, {max_val}]")
                    validated_item[feature] = float_value
                except (ValueError, TypeError):
                    raise ValueError(f"–ù–µ–∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–ª—è {feature}: {value}")
            else:
                validated_item[feature] = value
        
        validated_data.append(validated_item)
    
    return validated_data
```
